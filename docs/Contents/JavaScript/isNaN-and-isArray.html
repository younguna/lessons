isNaN and isArray

isNaN() isArray() 동작

const testInputs = [0, 1, 2, "1", "2",[1], ["a"], [], [1,"a"], [1,2,3], undefined, null]

testInputs.forEach((el)=>{
    console.log("original: ", el);
    console.log("!isNaN:   ", !isNaN(el));
    console.log("isArray:  ", isNaN(el));
    console.log("========================")
})
